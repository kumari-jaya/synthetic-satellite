Basic Usage


import { VortxAI } from '@vortx/ai-core';

const vortx = new VortxAI({
  apiKey: process.env.VORTX_API_KEY,
  version: '2025.1',
  region: 'auto'
});

// Modern async/await usage
async function analyzeData() {
  const result = await vortx.analyze({

  console.log(result.predictions);
}

// Stream responses
